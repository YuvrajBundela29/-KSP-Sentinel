   359â†’function simulateForce2D(
   360â†’  nodes: SimNode[],
   361â†’  links: SimLink[],
   362â†’  width: number,
   363â†’  height: number,
   364â†’  iterations: number = 300
   365â†’) {
   366â†’  const cx = width / 2;
   367â†’  const cy = height / 2;
   368â†’  const spread = Math.min(width, height) * 0.35;
   369â†’
   370â†’  for (const n of nodes) {
   371â†’    n.x = cx + (Math.random() - 0.5) * spread * 2;
   372â†’    n.y = cy + (Math.random() - 0.5) * spread * 2;
   373â†’    n.vx = 0;
   374â†’    n.vy = 0;
   375â†’  }
   376â†’
   377â†’  const nodeMap = new Map<string, SimNode>();
   378â†’  for (const n of nodes) nodeMap.set(n.id, n);
   379â†’
   380â†’  for (let iter = 0; iter < iterations; iter++) {
   381â†’    const alpha = 1 - iter / iterations;
   382â†’    const repulsion = 6000;
   383â†’    const attraction = 0.005;
   384â†’    const centerForce = 0.01;
   385â†’    const damping = 0.82;
   386â†’
   387â†’    // Repulsion between all pairs
   388â†’    for (let i = 0; i < nodes.length; i++) {
   389â†’      for (let j = i + 1; j < nodes.length; j++) {
   390â†’        const a = nodes[i];
   391â†’        const b = nodes[j];
   392â†’        let dx = a.x - b.x;
   393â†’        let dy = a.y - b.y;
   394â†’        let dist = Math.sqrt(dx * dx + dy * dy);
   395â†’        if (dist < 1) {
   396â†’          dist = 1;
   397â†’          dx = Math.random() - 0.5;
   398â†’          dy = Math.random() - 0.5;
   399â†’        }
   400â†’        const force = (repulsion * alpha) / (dist * dist);
   401â†’        const fx = (dx / dist) * force;
   402â†’        const fy = (dy / dist) * force;
   403â†’        a.vx += fx;
   404â†’        a.vy += fy;
   405â†’        b.vx -= fx;
   406â†’        b.vy -= fy;
   407â†’      }
   408â†’    }
   409â†’
   410â†’    // Attraction along links
   411â†’    for (const link of links) {
   412â†’      const s = nodeMap.get(link.source);
   413â†’      const t = nodeMap.get(link.target);
   414â†’      if (!s || !t) continue;
   415â†’      const dx = t.x - s.x;
   416â†’      const dy = t.y - s.y;
   417â†’      const dist = Math.sqrt(dx * dx + dy * dy);
   418â†’      if (dist < 1) continue;
   419â†’      const force = dist * attraction * alpha;
   420â†’      const fx = (dx / dist) * force;
   421â†’      const fy = (dy / dist) * force;
   422â†’      s.vx += fx;
   423â†’      s.vy += fy;
   424â†’      t.vx -= fx;
   425â†’      t.vy -= fy;
   426â†’    }
   427â†’
   428â†’    // Center gravity
   429â†’    for (const n of nodes) {
   430â†’      n.vx += (cx - n.x) * centerForce * alpha;
   431â†’      n.vy += (cy - n.y) * centerForce * alpha;
   432â†’    }
   433â†’
   434â†’    // Apply velocity with damping
   435â†’    for (const n of nodes) {
   436â†’      n.vx *= damping;
   437â†’      n.vy *= damping;
   438â†’      n.x += n.vx;
   439â†’      n.y += n.vy;
   440â†’    }
   441â†’  }
   442â†’}

   476â†’
   477â†’  // Canvas sizing

   501â†’

   504â†’    canvas.height = canvasSize.h * dpr;
   505â†’    ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
   506â†’
   507â†’    // Background
   508â†’    ctx.fillStyle = "#050810";

   510â†’
   511â†’    // Subtle grid
   512â†’    ctx.save();
   513â†’    ctx.strokeStyle = "rgba(255,255,255,0.015)";
   514â†’    ctx.lineWidth = 1;
   515â†’    const gridSize = 40 * zoom;
   516â†’    const offsetX = (panX % gridSize + gridSize) % gridSize;
   517â†’    const offsetY = (panY % gridSize + gridSize) % gridSize;
   518â†’    for (let x = offsetX; x < canvasSize.w; x += gridSize) {
   519â†’      ctx.beginPath();
   520â†’      ctx.moveTo(x, 0);
   521â†’      ctx.lineTo(x, canvasSize.h);
   522â†’      ctx.stroke();
   523â†’    }
   524â†’    for (let y = offsetY; y < canvasSize.h; y += gridSize) {
   525â†’      ctx.beginPath();
   526â†’      ctx.moveTo(0, y);
   527â†’      ctx.lineTo(canvasSize.w, y);
   528â†’      ctx.stroke();
   529â†’    }
   530â†’    ctx.restore();
   531â†’
   532â†’    // Transform
   533â†’    ctx.save();
   534â†’    ctx.translate(canvasSize.w / 2 + panX, canvasSize.h / 2 + panY);
   535â†’    ctx.scale(zoom, zoom);
   536â†’    ctx.translate(-canvasSize.w / 2, -canvasSize.h / 2);
