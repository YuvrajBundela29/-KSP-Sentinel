   107вЖТ   107вЖТfunction useAnimatedCounter(target: number, duration = 1200) {
   108вЖТ   108вЖТ  const [count, setCount] = useState(0);
   109вЖТ   109вЖТ  useEffect(() => {
   110вЖТ   110вЖТ    let start = 0;
   111вЖТ   111вЖТ    const step = target / (duration / 16);
   112вЖТ   112вЖТ    const timer = setInterval(() => {
   113вЖТ   113вЖТ      start += step;
   114вЖТ   114вЖТ      if (start >= target) {
   115вЖТ   115вЖТ        setCount(target);
   116вЖТ   116вЖТ        clearInterval(timer);
   117вЖТ   117вЖТ      } else {
   118вЖТ   118вЖТ        setCount(Math.floor(start));
   119вЖТ   119вЖТ      }
   120вЖТ   120вЖТ    }, 16);
   121вЖТ   121вЖТ    return () => clearInterval(timer);
   122вЖТ   122вЖТ  }, [target, duration]);
   123вЖТ   123вЖТ  return count;
   124вЖТ   124вЖТ}
